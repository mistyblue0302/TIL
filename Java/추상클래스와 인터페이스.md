```java
public class Calculator implements Calculatable {
    int a,b;
    public void set(int a, int b) {
	this.a = a;
	this.b = b;
    }
    public int sum() {
	return this.a + this.b;
    }
    public int avg() {
	return (this.a + this.b)/2;
    }
}
```
